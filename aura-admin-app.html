} const updatesSheet = ss.getSheetByName('driver_updates'); if (updatesSheet.getLastRow() === 0) { updatesSheet.appendRow(UPDATES_HEADERS); updatesSheet.getRange(1, 1, 1, UPDATES_HEADERS.length) .setFontWeight('bold').setBackground('#1D9E75').setFontColor('#ffffff'); } let ordersSheet = ss.getSheetByName('orders'); if (!ordersSheet) { ordersSheet = ss.insertSheet('orders'); ordersSheet.appendRow(ORDERS_HEADERS); ordersSheet.getRange(1, 1, 1, ORDERS_HEADERS.length) .setFontWeight('bold').setBackground('#1D9E75').setFontColor('#ffffff'); } return 'Setup hoàn tất!'; } // ============================================================ // doGet // ============================================================ function doGet(e) { const action = e.parameter.action || ''; } else if (action === 'assignDriver') { result = assignDriver(body); } else if (action === 'cancelOrder') { result = cancelOrder(body); } else { result = { success: false, message: 'Action không hợp lệ' }; } return jsonResponse(result); } catch (err) { return jsonResponse({ success: false, message: err.toString() }); } } // ============================================================ // DRIVER APP FUNCTIONS // ============================================================ function loginDriver(code) { if (!code) return { success: false, message: 'Thiếu mã tài xế' }; const nameIdx = headers.indexOf('driver_name'); const statusIdx = headers.indexOf('status'); for (let i = 1; i < data.length; i++) { if (String(data[i][codeIdx]).trim() === String(code).trim()) { const status = data[i][statusIdx]; if (!sameDriver) continue; if (!includeAll && obj.status === 'done') continue; try { obj.stops = JSON.parse(obj.stops || '[]'); } catch(e) { obj.stops = []; } orders.push({ id: obj.order_id, date: obj.date ? String(obj.date) : '', stops: obj.stops, status: obj.status, note: obj.note || '', created_at: obj.created_at ? String(obj.created_at) : '' }); } // Đơn active lên đầu, đơn done xuống cuối (mới nhất lên trước trong done) orders.sort(function(a, b) { if (a.status === 'done' && b.status !== 'done') return 1; if (a.status !== 'done' && b.status === 'done') return -1; return { success: true, message: 'Đã lưu cập nhật' }; } function finishOrder(body) { const ss = SpreadsheetApp.openById(SHEET_ID); const updatesSheet = ss.getSheetByName('driver_updates'); updatesSheet.appendRow([ new Date(), body.driver_code, body.driver_name, body.order_id, 'FINISH', 'finish', 'Kết thúc đơn', '', 'done', 'Hoàn thành: ' + (body.done_count || 0) + ', Thất bại: ' + (body.fail_count || 0), '' ]); // Cập nhật trạng thái đơn thành done const ordersSheet = ss.getSheetByName('orders'); if (ordersSheet) { const data = ordersSheet.getDataRange().getValues(); const headers = data[0]; const idIdx = headers.indexOf('order_id'); const statusIdx = headers.indexOf('status'); for (let i = 1; i < data.length; i++) { if (String(data[i][idIdx]).trim() === String(body.order_id).trim()) { ordersSheet.getRange(i + 1, statusIdx + 1).setValue('done'); break; } } } return { success: true, message: 'Đơn đã kết thúc' }; } // ============================================================ // ADMIN APP FUNCTIONS // ============================================================ function getAdminOrders() { const ss = SpreadsheetApp.openById(SHEET_ID); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { success: true, drivers: [] }; const headers = data[0]; const drivers = []; for (let i = 1; i < data.length; i++) { const row = data[i]; let sheet = ss.getSheetByName('orders'); if (!sheet) { sheet = ss.insertSheet('orders'); sheet.appendRow(ORDERS_HEADERS); sheet.getRange(1, 1, 1, ORDERS_HEADERS.length) .setFontWeight('bold').setBackground('#1D9E75').setFontColor('#ffffff'); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const idIdx = headers.indexOf('order_id'); for (let i = 1; i < data.length; i++) { if (String(data[i][idIdx]).trim() === String(body.order_id).trim()) { return { success: false, message: 'Mã đơn đã tồn tại' }; } } sheet.appendRow([ body.order_id, body.date || '', body.driver_code || '', body.driver_name || '', 'pending', body.note || '', JSON.stringify(body.stops || []), new Date() ]); return { success: true, message: 'Tạo đơn thành công' }; const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName('orders'); if (!sheet) return { success: false, message: 'Không tìm thấy sheet orders' }; const data = sheet.getDataRange().getValues(); const headers = data[0]; const idIdx = headers.indexOf('order_id'); const statusIdx = headers.indexOf('status'); for (let i = 1; i < data.length; i++) { if (String(data[i][idIdx]).trim() === String(body.order_id).trim()) { const currentStatus = data[i][statusIdx]; return { success: true, message: 'Đã hủy đơn ' + body.order_id }; } } return { success: false, message: 'Không tìm thấy đơn hàng' }; } // ============================================================ // HELPER // ============================================================ function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON);
